-- Create Database
CREATE DATABASE IF NOT EXISTS triptales_db;
USE triptales_db;

-- Create Users Table
CREATE TABLE IF NOT EXISTS users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100) NOT NULL,
  email VARCHAR(100) NOT NULL UNIQUE,
  username VARCHAR(50) NOT NULL UNIQUE,
  password VARCHAR(255) NOT NULL,
  gender VARCHAR(10),
  dob DATE,
  country VARCHAR(100),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE travel_logs (
  id INT AUTO_INCREMENT PRIMARY KEY,
  location VARCHAR(255) NOT NULL,
  experience TEXT NOT NULL,
  budget DECIMAL(10,2) NOT NULL,
  suggestions TEXT,
  photos JSON NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


-- Posts Table
CREATE TABLE posts (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT,
  title VARCHAR(255) NOT NULL,
  location_name VARCHAR(255) NOT NULL,
  latitude DOUBLE,
  longitude DOUBLE,
  experience TEXT NOT NULL,
  budget DECIMAL(10,2) NOT NULL,
  duration_days INT,
  best_season VARCHAR(100),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
  -- FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL -- only if `users` table exists
);


-- Post Images Table
CREATE TABLE post_images (
  id INT AUTO_INCREMENT PRIMARY KEY,
  post_id INT NOT NULL,
  image_url VARCHAR(255) NOT NULL,
  caption VARCHAR(255),
  FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE
);

During our 3-day trip to Manali, we stayed at a cozy riverside hotel with spectacular mountain views. The hotel staff was friendly, and the rooms were clean and comfortable, though a bit small. We enjoyed delicious local Himachali cuisine, especially the trout fish and siddu at nearby eateries. The trip included thrilling activities like river rafting and a trek to the Solang Valley. Weather was pleasant throughout, making sightseeing very enjoyable. The budget of ₹15,000 covered accommodation, food, and activities comfortably. We highly recommend trying the local apple cider and shopping at the Mall Road market for handicrafts.
